            <label class="block text-sm font-medium text-gray-700 mb-1">鍍膜 (Coating)</label>

            <select id="coating" class="w-full p-2.5 border border-gray-300 rounded-lg bg-white shadow-xs focus:ring-2 focus:ring-blue-500 focus:outline-hidden">

                <option value="沒有">沒有鍍膜</option>

                <option value="抗刮">抗刮 (HC)</option>

                <option value="抗刮+超透光">抗刮 + 超透光 (HMC)</option>

            </select>

        <div class="mb-6">

            <label class="block text-sm font-medium text-gray-700 mb-1">顏色</label>

            <select id="color" class="w-full p-2.5 border border-gray-300 rounded-lg bg-white shadow-xs focus:ring-2 focus:ring-blue-500 focus:outline-hidden">

                <!-- 由 JavaScript 動態動態生成 -->

            </select>

        <!-- 查詢按鈕 -->

        <button onclick="lookupCode()" class="w-full bg-blue-600 hover:bg-blue-700 text-white font-semibold p-3 rounded-lg transition duration-200 shadow-md">

            查詢鏡片編號

        </button>


        <!-- 結果顯示區域 -->

        <div class="mt-6 p-4 bg-gray-100 rounded-xl text-center hidden" id="resultBox">

            <p class="text-sm text-gray-500 font-medium">對應 HKO 鏡片編號</p>

            <p class="text-3xl font-black text-gray-900 mt-1 tracking-wider" id="resultCode">------</p>

        </div>

    <script>

        // 顏色聯動關係

        const colorMap = {

            "染色鏡片": ["灰", "藍"],

            "腮紅鏡片": ["紅", "橙"],

            "Photoshade": ["灰", "棕"]

        };


        // 模擬的 HKO 編號資料庫 (你可以根據實際編號修改這裡的 Mapping)

        // 格式： "類型-折射率-鍍膜-顏色": "實際編號"

        const hkoDatabase = {

            "染色鏡片-1.6-抗刮-灰": "HKO60T-GY-H",

            "染色鏡片-1.67-抗刮+超透光-藍": "HKO67T-BL-HMC",

            "腮紅鏡片-1.6-沒有-紅": "HKO60B-RD-UC",

            "Photoshade-1.67-抗刮+超透光-棕": "HKO67P-BR-HMC",

            // ... 在這裡自由添加你的對應數據

        };


        // 根據選擇的鏡片類型，自動更新顏色選項

        function updateColors() {

            const type = document.getElementById("lensType").value;

            const colorSelect = document.getElementById("color");

            

            // 清空舊選項

            colorSelect.innerHTML = "";

            

            // 插入新選項

            colorMap[type].forEach(color => {

                const opt = document.createElement("option");

                opt.value = color;

                opt.textContent = color;

                colorSelect.appendChild(opt);

            });

        }


        // 執行查詢

        function lookupCode() {

            const type = document.getElementById("lensType").value;

            const idx = document.getElementById("index").value;

            const coating = document.getElementById("coating").value;

            const color = document.getElementById("color").value;


            // 組裝用來查詢的 Key

            const searchKey = `${type}-${idx}-${coating}-${color}`;

            

            // 從資料庫找編號，找不到就自動動態生成一個臨時編號

            let finalCode = hkoDatabase[searchKey];

            

            if (!finalCode) {

                // 臨時生成的邏輯例子 (若資料庫沒精準配對)

                finalCode = `HKO-${idx}-${color}`; 

            }


            // 顯示結果

            document.getElementById("resultCode").textContent = finalCode;

            document.getElementById("resultBox").classList.remove("hidden");

        }


        // 網頁載入時先初始化一次顏色選單

        window.onload = updateColors;

    </script>
